3   /**
4    * Strategy for buttons on calculator represening functions.
5    * Function represents action where only current number on screen is used
6    * and replaced by result of function. <br>
7    * Function buttons with inverse function following: <br>
8    * -log (10^) <br>
9    * -ln (e^) <br>
10   * -cos (arccos) <br>
11   * -sin (arcsin)<br>
12   * -ctg (arcctg)<br>
13   * -tg (arctg)<br>
14   * -1/x <br>
15   * -x^n (x^1/n) <br>
16   * @author Teo Toplak
17   *
18   */
19  public interface IFunction {
20  	/**
21  	 * Default function of button.
22  	 * @param num argument number (from screen)
23  	 * @return result of function operation
24  	 */
25  	public double function(double num);
26  	/**
27  	 * Inversed function of button.
28  	 * @param num argument number (from screen)
29  	 * @return result of function operation
30  	 */
31  	public double invFunction(double num);
32  }
